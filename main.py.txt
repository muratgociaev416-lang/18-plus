import asyncio
import logging
import random
import os
from aiogram import Bot, Dispatcher, types, F
from aiogram.filters import Command
from aiogram.types import FSInputFile, CallbackQuery

from config import BOT_TOKEN, ADMIN_ID, CATEGORIES, MEDIA_PATH
from keyboards import get_main_menu, get_after_media
from database import load_users, save_users
from texts import get_erotic_text
from keep_alive import keep_alive

bot = Bot(token=BOT_TOKEN, parse_mode="HTML")
dp = Dispatcher()
users = load_users()

# ======================= СТАРТ =======================
@dp.message(Command("start"))
async def start(message: types.Message):
    user_id = str(message.from_user.id)
    if user_id not in users:
        users[user_id] = {"age_confirmed": False, "favorites": [], "stats": 0}
        save_users(users)
    
    if not users[user_id]["age_confirmed"]:
        kb = [[types.InlineKeyboardButton(text="✅ Мне уже 18+", callback_data="confirm_age")]]
        await message.answer(
            "🔞 <b>ВНИМАНИЕ! КОНТЕНТ 18+</b>\n\n"
            "Если тебе меньше 18 лет — немедленно покинь этот бот.",
            reply_markup=types.InlineKeyboardMarkup(inline_keyboard=kb)
        )
    else:
        await message.answer("🔥 С возвращением, грязный мальчик 😉", reply_markup=get_main_menu())


@dp.callback_query(F.data == "confirm_age")
async def confirm_age(callback: CallbackQuery):
    user_id = str(callback.from_user.id)
    users[user_id]["age_confirmed"] = True
    save_users(users)
    await callback.message.edit_text("✅ Доступ открыт. Наслаждайся!", reply_markup=get_main_menu())


@dp.callback_query(F.data == "main_menu")
async def back_to_menu(callback: CallbackQuery):
    await callback.message.edit_text("🔥 Главное меню", reply_markup=get_main_menu())


# ======================= КАТЕГОРИИ =======================
@dp.callback_query(F.data.startswith("cat_"))
async def show_category(callback: CallbackQuery):
    cat = callback.data.replace("cat_", "")
    await send_random_media(callback, cat)


async def send_random_media(callback: CallbackQuery, category: str, is_more=False):
    folder = f"{MEDIA_PATH}/{category}"
    if not os.path.exists(folder):
        return await callback.answer("⛔ Папка не найдена", show_alert=True)

    files = [f for f in os.listdir(folder) if f.lower().endswith(('.jpg','.jpeg','.png','.mp4','.gif'))]
    if not files:
        return await callback.answer("📭 В этой категории пока пусто", show_alert=True)

    file_name = random.choice(files)
    file_path = f"{folder}/{file_name}"

    user_id = str(callback.from_user.id)
    users[user_id]["stats"] += 1
    save_users(users)

    try:
        caption = get_erotic_text()
        if file_name.lower().endswith(('.mp4', '.gif')):
            await callback.message.answer_video(
                FSInputFile(file_path), caption=caption,
                reply_markup=get_after_media(category, file_name)
            )
        else:
            await callback.message.answer_photo(
                FSInputFile(file_path), caption=caption,
                reply_markup=get_after_media(category, file_name)
            )
    except Exception as e:
        logging.error(e)
        await callback.answer("❌ Ошибка отправки файла")


@dp.callback_query(F.data.startswith("more_"))
async def more_content(callback: CallbackQuery):
    category = callback.data.replace("more_", "")
    await send_random_media(callback, category, True)


@dp.callback_query(F.data.startswith("like_"))
async def add_to_favorites(callback: CallbackQuery):
    _, category, filename = callback.data.split("_", 2)
    user_id = str(callback.from_user.id)
    fav = {"category": category, "file": filename}
    
    if fav not in users[user_id]["favorites"]:
        users[user_id]["favorites"].append(fav)
        save_users(users)
        await callback.answer("❤️ Добавлено в избранное!", show_alert=True)
    else:
        await callback.answer("Уже есть в избранном")


# ======================= ОСТАЛЬНЫЕ ФУНКЦИИ =======================
@dp.callback_query(F.data == "favorites")
async def show_favorites(callback: CallbackQuery):
    user_id = str(callback.from_user.id)
    favs = users.get(user_id, {}).get("favorites", [])
    if not favs:
        return await callback.answer("❤️ Избранное пустое", show_alert=True)
    await callback.message.answer(f"❤️ В избранном: {len(favs)} файлов")


@dp.callback_query(F.data == "profile")
async def profile(callback: CallbackQuery):
    user_id = str(callback.from_user.id)
    u = users.get(user_id, {})
    await callback.message.answer(
        f"👤 <b>Профиль</b>\n\n"
        f"Просмотрено: {u.get('stats', 0)}\n"
        f"В избранном: {len(u.get('favorites', []))}\n"
        f"Статус: 18+ ✅"
    )


@dp.message(Command("admin"))
async def admin_panel(message: types.Message):
    if message.from_user.id != ADMIN_ID:
        return
    await message.answer(f"👑 Админ панель\nПользователей: {len(users)}")


# ======================= ЗАПУСК =======================
async def main():
    logging.basicConfig(level=logging.INFO)
    asyncio.create_task(keep_alive())
    
    print("🚀 Максимальный 18+ бот запущен на Replit!")
    print("🌐 Анти-сон активен")
    
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
